6   package com.mipyykko.muistipeli.malli;

8   /**
9    *
10   * @author pyykkomi

12  public class Korttitehdas {
13      
14      private String korttityyppi;
15      
16      public Korttitehdas(String korttityyppi) {
17          this.korttityyppi = korttityyppi;
18      }
19  
20      public Kortti uusiKortti(Object... params) {
21          switch (korttityyppi) {
22              case "Geneerinen":
23                  return new GeneerinenKortti((Kuva) params[0], (Tausta) params[1]);
24              case "JavaFX":
25                  return new JavaFXKortti((Kuva) params[0], (Tausta) params[1]);
26              default:
27                  return null;
28          }
29      }
30  }
