6   package com.mipyykko.muistipeli.malli;
7   
8   import com.mipyykko.muistipeli.malli.enums.Korttityyppi;
9   import com.mipyykko.muistipeli.malli.impl.TekstiKortti;
10  import com.mipyykko.muistipeli.malli.impl.JavaFXKortti;
11  
12  /**
13   * Konstruktorille annetaan tehtaan haluttu korttityyppi.
14   * 
15   * @author pyykkomi
16   */
17  public class Korttitehdas {
18      
19      private final Korttityyppi korttityyppi;
20      
21      public Korttitehdas(Korttityyppi korttityyppi) {
22          this.korttityyppi = korttityyppi;
23      }
24  
25      /**
26       * Uuden korttityypin kortin luova ja palauttava metodi.
27       * 
28       * @param params Vaihteleva määrä kortin konstruktorille annettavia parametreja
29       * @return korttityypin mukainen Kortti tai null jos korttityyppi on väärä
30       */
31      public Kortti uusiKortti(Object... params) {
32          if (korttityyppi == Korttityyppi.TEKSTI) {
33              return new TekstiKortti((Kuva) params[0], (Tausta) params[1]);
34          } else if (korttityyppi == Korttityyppi.JAVAFX) {
35              return new JavaFXKortti((Kuva) params[0], (Tausta) params[1]);
36          }
37          return null;
38      }
39  }
