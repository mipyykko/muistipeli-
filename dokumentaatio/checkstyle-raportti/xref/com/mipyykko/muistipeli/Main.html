40          UITyyppi uiTyyppi = UITyyppi.JAVAFX;
41          Korttityyppi korttityyppi = Korttityyppi.JAVAFX;
42  
43          switch (uiTyyppi) {
44              case TEKSTI:
45                  Set<Kuva> testikuvat = new HashSet<>();
46                  Set<Tausta> testitaustat = new HashSet<>();
47  
48                  int leveys = 4;
49                  int korkeus = 4;
50  
51                  for (int i = 0; i < (leveys * korkeus) / 2; i++) {
52                      testikuvat.add(new TekstiKuva(Integer.toString(i + 1)));
53                      testitaustat.add(new TekstiTausta("*"));
54                      testitaustat.add(new TekstiTausta("*"));
55                  }
56  
57                  Peli peli = new Peli(null, korttityyppi);
58                  try {
59                      peli.uusiPeli(leveys, korkeus, testikuvat, testitaustat);
60                  } catch (Exception ex) {
61                      System.out.println("hetkinen?"); //TODO
62                  }
63                  UI ui = new TekstiUI(peli, new Scanner(System.in));
64                  peli.setUI(ui);
65                  ui.nayta();
66                  break;
67              case JAVAFX:
68                  Application.launch(JavaFXMain.class, args);
69                  break;
70              default:
71                  throw new AssertionError();
72          }
73      }
