12  import com.mipyykko.muistipeli.malli.enums.UITyyppi;

17  import java.util.Set;
18  import javafx.application.Application;
19  
20  /** 
21   * Muistipeliö, simppeliäkin simppelimpi javalabra-projekti.
22   * 
23   * @author pyykkomi
24   */
25  public class Main {
26  
27      /**
28       * Pääluokka. JavaFX haarautuu tästä omaansa.
29       * @param args komentoriviparametrit - ainoa toistaiseksi toimiva on
30       * "teksti" joka käynnistää pelin tekstimuoodissa.
31       */
32      public static void main(String[] args) {
33  
34          UITyyppi uiTyyppi = UITyyppi.JAVAFX;
35          Korttityyppi korttityyppi = Korttityyppi.JAVAFX;
36  
37          if (args.length > 0) {
38              String s = args[0].toLowerCase();
39              if (s.equals("teksti")) {
40                  uiTyyppi = UITyyppi.TEKSTI;
41                  korttityyppi = Korttityyppi.TEKSTI;
42              }
43          }
44  
45          switch (uiTyyppi) {
46              case TEKSTI:
47  
48                  int leveys = 4;
49                  int korkeus = 4;
50  
51                  TekstiInit ti = new TekstiInit();
52                  
53                  Set<Kuva> kuvat = ti.luoKuvat(leveys, korkeus);
54                  Set<Tausta> taustat = ti.luoTaustat(leveys, korkeus);
55  
56                  Peli peli = new Peli(korttityyppi);
57                  try {
58                      peli.uusiPeli(leveys, korkeus, kuvat, taustat);
59                  } catch (Exception ex) {
60                      System.out.println("hetkinen?"); //TODO
61                  }
62                  UI ui = new TekstiUI(peli, new Scanner(System.in));
63                  try {
64                      ui.nayta();
65                  } catch (Exception e) {
66                  }
67                  break;
68              case JAVAFX:
69                  Application.launch(JavaFXMain.class, args);
70                  break;
71              default:
72                  throw new AssertionError();
73          }
74      }
75  }
