46      /**
47       * Luo kuvalistan jossa oikeat osoitteet resursseihin.
48       * 
49       * @param tileset Hakemiston nimi. Metodi odottaa hakemistossa olevan tileset.txt-nimisen tiedoston.
50       */
51      public void lueKuvalista(String tileset) {
52          kuvalista = new HashMap<>();
53  
54          InputStream is = Main.class.getClassLoader().getResourceAsStream("kuvat/" + tileset + "/tileset.txt");
55          try (Scanner s = new Scanner(is)) {
56              while (s.hasNextLine()) {
57                  String[] t = s.nextLine().split(",");
58                  kuvalista.put(t[0].replaceAll("^\"|\"$", ""),
59                          "kuvat/" + tileset + "/images/" + t[1].replaceAll("^\"|\"$", ""));
60              }
61          }
62      }
63      /**
64       * Luo Image-objektit korttien kuville.
65       */
66      public void luoKuvat() {
67          Iterator<String> it = kuvalista.keySet().iterator();
68          double maxWidth = 0;
69          double maxHeight = 0;
70          
71          for (int i = 0; i < (leveys * korkeus) / 2; i++) {
72              String key = it.next();
73              // TODO: kuvien skaalaus pois?
74              Image kuva = new Image(getClass().getClassLoader().getResource(kuvalista.get(key)).toString(), 200, 200, true, false);
75              maxWidth = Math.max(maxWidth, kuva.getWidth());
76              maxHeight = Math.max(maxHeight, kuva.getHeight());
77              kuvat.add(new JavaFXKuva(key, kuva));
78              // test
79              Image tausta = new Image(getClass().getClassLoader().getResource("taustat/basic/tausta.png").toString(), 200, 200, true, false);
80              maxWidth = Math.max(maxWidth, tausta.getWidth());
81              maxHeight = Math.max(maxHeight, tausta.getHeight());
82              taustat.add(new JavaFXTausta("tausta", tausta));
83              taustat.add(new JavaFXTausta("tausta", tausta));
84          }
85  
86          
87      }
