46      /**
47       * Luo valikkoikkunan ja valikon sisältöineen.
48       *
49       */
50      public ValikkoIkkuna()/* throws Exception*/ {
51          super();
52          this.primaryStage = JavaFXMain.getStage();
53          sisalto = new GridPane();
54          setAlignment(Pos.CENTER);
55          setPadding(new Insets(15));
56          sisalto.setHgap(16);
57          sisalto.setVgap(8);
58          setBackground(new Background(new BackgroundFill(Color.YELLOW, null, null)));
59  
60          sisalto.setBackground(null);
61          Text otsikko = new Text("Muistipeliö");
62          otsikko.setFont(Font.loadFont(getClass().getClassLoader().getResourceAsStream("fontit/GoodDog.otf"), 48));
63          otsikko.setTextAlignment(TextAlignment.CENTER);
64  
65          aloitusnappi = new Button("Aloita peli");
66          aloitusnappi.setAlignment(Pos.CENTER);
67          aloitusnappi.setOnAction((ActionEvent event) -> {
68              //try {
69              aloitusNappiKlikattu(event);
70              //} catch (Exception ex) {
71              //    System.err.println("Jotain tapahtui aloitusnappia painettassa :x " + ex.getMessage()); // DEBUG
72              //}
73          });
74  
75          kuvavalikko = new ComboBox();
76          //debug:
77          ObservableList<String> kuvaOptions = FXCollections.observableArrayList();
78          try {
79              for (String s : new JavaFXInit().haeKuvasetit()) {
80                  String[] hakemisto = s.split("/");
81                  kuvaOptions.add(hakemisto[hakemisto.length - 1]);
82              }
83          } catch (Exception e) {
84              //throw new Exception("Ei kuvasettejä odotetussa hakemistossa!");
85          }
86          kuvavalikko.setItems(kuvaOptions);
87          kuvavalikko.getSelectionModel().selectFirst();
88  
89          sisalto.add(otsikko, 1, 0);
90          sisalto.add(kuvavalikko, 1, 1);
91          sisalto.add(aloitusnappi, 1, 2);
92  
93          GridPane.setHalignment(otsikko, HPos.CENTER);
94          GridPane.setHalignment(aloitusnappi, HPos.CENTER);
95          GridPane.setHalignment(kuvavalikko, HPos.CENTER);
96  
97          add(sisalto, 1, 1);
98      }

113     private void aloitusNappiKlikattu(ActionEvent event)/* throws Exception*/ {
114         //if (peli == null || peli.getTila() == Pelitila.VALIKKO) {
115         int leveys = 4; // DEBUG, TODO jne. jne.
116         int korkeus = 5;
117         JavaFXInit jfi = new JavaFXInit();
118         String setti = (String) kuvavalikko.getValue();
119         if (setti == null || setti.isEmpty()) {
120             return;
121         }
122         jfi.lueKuvalista(setti);
123         jfi.lueTaustalista("debug");
124         Set<Kuva> kuvat = null;
125         try {
126             kuvat = jfi.luoKuvat(leveys, korkeus);
127         } catch (Exception ex) { // debug
128             Logger.getLogger(ValikkoIkkuna.class.getName()).log(Level.SEVERE, null, ex);
129         }
130         Set<Tausta> taustat = null;
131         try {
132             taustat = jfi.luoTaustat(leveys, korkeus);
133         } catch (Exception ex) { // debug
134             Logger.getLogger(ValikkoIkkuna.class.getName()).log(Level.SEVERE, null, ex);
135         }
136         // debug pelin käynnistys?
137         Peli peli = new Peli(Korttityyppi.JAVAFX);
138         try {
139             peli.uusiPeli(leveys, korkeus, kuvat, taustat);
140             PeliIkkuna peliIkkuna = new PeliIkkuna(peli);
141             //peliIkkuna.getRuudukko().alustaRuudukko();
142             primaryStage.setScene(new Scene(peliIkkuna, this.getScene().getWidth()/*ikkunaleveys*/,
143                     this.getScene().getHeight()/*ikkunakorkeus*/));
144         } catch (Exception e) {
145             System.out.println("Pelin luominen epäonnistui! " + e.getMessage()); // debug
146         }
147     }
