33  public class ValikkoController implements Initializable, ControlledRuutu {
34  
35      private IkkunaController ikkunaController;
36      
37      @FXML private Text otsikko;
38      @FXML private ComboBox kuvavalikko;
39      @FXML private Button aloitusnappi;
40      
41      @Override
42      public void initialize(URL location, ResourceBundle resources) {
43          otsikko.setFont(Font.loadFont(getClass().getResourceAsStream("/fontit/GoodDog.otf"), 48));
44          ObservableList<String> kuvaOptions = FXCollections.observableArrayList();
45          try {
46              for (String s : new JavaFXInit().haeKuvasetit()) {
47                  String[] hakemisto = s.split("/");
48                  kuvaOptions.add(hakemisto[hakemisto.length - 1]);
49              }
50          } catch (Exception e) {
51              //throw new Exception("Ei kuvasettej√§ odotetussa hakemistossa!");
52          }
53          kuvavalikko.setItems(kuvaOptions);
54          kuvavalikko.getSelectionModel().selectFirst();
55          kuvavalikko.setLayoutX(0);
56          
57          aloitusnappi.setOnAction((ActionEvent ae) -> aloitusNappiKlikattu(ae));
58      }
59  
60      @Override
61      public void asetaParent(IkkunaController parent) {
62          ikkunaController = parent;
63      }
64      
65      private void aloitusNappiKlikattu(ActionEvent event)/* throws Exception*/ {
66          //if (peli == null || peli.getTila() == Pelitila.VALIKKO) {
67          int leveys = 4; // DEBUG, TODO jne. jne.
68          int korkeus = 5;
69          JavaFXInit jfi = new JavaFXInit();
70          String setti = (String) kuvavalikko.getValue();
71          if (setti == null || setti.isEmpty()) {
72              return;
73          }
74          jfi.lueKuvalista(setti);
75          jfi.lueTaustalista("debug");
76          Set<Kuva> kuvat = null;
77          try {
78              kuvat = jfi.luoKuvat(leveys, korkeus);
79          } catch (Exception ex) { // debug
80              System.out.println(""); 
81          }
82          Set<Tausta> taustat = null;
83          try {
84              taustat = jfi.luoTaustat(leveys, korkeus);
85          } catch (Exception ex) { // debug
86              System.out.println("");
87          }
88          Peli peli = new Peli(Korttityyppi.JAVAFX);
89          try {
90              peli.uusiPeli(leveys, korkeus, kuvat, taustat);
91              ikkunaController.setPeli(peli);
92              ((PeliController) ikkunaController.getController(JavaFXIkkuna.PELI)).alustaRuudukko();
93              ikkunaController.asetaIkkuna(JavaFXIkkuna.PELI);
94          } catch (Exception e) {
95              System.out.println("Pelin luominen ep√§onnistui! " + e.getMessage()); // debug
96          }
97      }
