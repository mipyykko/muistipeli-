16  /**
17   * JavaFX-käyttöliittymä.
18   *
19   * @author pyykkomi
20   */
21  public class JavaFXUI implements UI {
22  
23      private Peli peli;
24      private int ikkunaleveys, ikkunakorkeus;
25      private Stage primaryStage;
26      private Group root;
27      private StackPane paaIkkuna;
28      private Scene scene;
29      private boolean firstrun;
30  
31      /**
32       * JavaFXUI:n konstruktori.
33       *
34       */
35      public JavaFXUI() {
36          this.peli = null;
37          this.ikkunaleveys = 600; // magic numbers!
38          this.ikkunakorkeus = 600;
39          this.firstrun = true;
40  
41      }
42  
43      public void setStage(Stage stage) {
44          this.primaryStage = stage;
45      }
46  
47      @Override
48      public void setPeli(Peli peli) {
49          this.peli = peli;
50      }
51  
52      /**
53       * Näytä pelilauta.
54       */
55      @Override
56      public void nayta() /*throws Exception*/ {
57          if (firstrun) {
58              firstrun = false;
59              //try {
60              // TODO: koko ikkuna stackpane, johon lisätään ja josta poistetaan tasoja
61              IkkunaController ic = new IkkunaController();
62              ic.lataaIkkuna(JavaFXIkkuna.VALIKKO);
63              ic.lataaIkkuna(JavaFXIkkuna.PELI);
64              ic.lataaIkkuna(JavaFXIkkuna.TULOS);
65              
66              ic.asetaIkkuna(JavaFXIkkuna.VALIKKO);
67             
68              root = new Group();
69              root.getChildren().add(ic);
70  //            valikkoIkkuna = new ValikkoIkkuna();
71              
72              if (peli != null) {
73                  peli.setTila(Pelitila.VALIKKO);
74              }
75  
76              ic.prefWidthProperty().bind(primaryStage.widthProperty());
77              ic.prefHeightProperty().bind(primaryStage.heightProperty());
78              scene = new Scene(root/*valikkoIkkuna*/, ikkunaleveys, ikkunakorkeus);
79              scene.setFill(Color.YELLOW);
80  
81              primaryStage.setTitle("Muistipeliö");
82              primaryStage.setScene(scene);
83              primaryStage.sizeToScene();
84              primaryStage.show();
85              //} catch (Exception e) {
86              //    throw new Exception("UI:n alustus ei onnistunut: " + e.getMessage());
87              //}
88          }
89      }
90  
91      private void sulje() {
92          primaryStage.close();
93      }
94  
95  }
