6   package com.mipyykko.muistipeli.malli;
7   
8   import java.util.Objects;
9   
10  /**
11   *
12   * @author pyykkomi
13   */
14  public class Kortti implements Comparable<Kortti> {
15      private Kuva kuva;
16      private Tausta tausta;
17      private boolean kaannetty;
18      
19      public Kortti(Kuva kuva, Tausta tausta) {
20          this.kuva = kuva;
21          this.tausta = tausta;
22          this.kaannetty = false;
23      }
24  
25      public boolean kaannetty() {
26          return kaannetty;
27      }
28      
29      public boolean kaanna() {
30          kaannetty = !kaannetty; // en tiiä pitäiskö tähän tehdä flipiti flip
31          return kaannetty;
32      }
33      
34      public Kuva getKuva() {
35          return kuva;
36      }
37  
38      public void setKuva(Kuva kuva) {
39          this.kuva = kuva;
40      }
41  
42      public Tausta getTausta() {
43          return tausta;
44      }
45  
46      public void setTausta(Tausta tausta) {
47          this.tausta = tausta;
48      }
49      
50      @Override
51      public int compareTo(Kortti k) {
52          return this.kuva.toString().compareTo(k.kuva.toString());
53      }
54      
55      //@Override
56      public boolean equals(Kortti k) {
57          return this.kuva.toString().equals(k.kuva.toString());
58      }
59  }
